10 Pseudocodes

1. BUBBLE SORT
FOR i = 0 TO n-2
  FOR j = 0 TO n-2-i
    IF A[j] > A[j+1] SWAP A[j], A[j+1]

2. SELECTION SORT
FOR i = 0 TO n-2
  min = i
  FOR j = i+1 TO n-1
    IF A[j] < A[min] min = j
  SWAP A[i], A[min]

3. INSERTION SORT
FOR i = 1 TO n-1
  key = A[i]; j = i-1
  WHILE j >= 0 AND A[j] > key
    A[j+1] = A[j]; j = j-1
  A[j+1] = key

4. MERGE SORT
FUNCTION mergesort(A)
  IF len(A) <= 1 RETURN A
  mid = len(A)//2
  L = mergesort(A[:mid]); R = mergesort(A[mid:])
  RETURN merge(L,R)

5. QUICK SORT
FUNCTION quicksort(A, lo, hi)
  IF lo < hi
    p = partition(A, lo, hi)
    quicksort(A, lo, p-1)
    quicksort(A, p+1, hi)

6. LINEAR SEARCH
FOR i = 0 TO n-1
  IF A[i] == key RETURN i
RETURN -1

7. BINARY SEARCH (iterative)
lo=0; hi=n-1
WHILE lo <= hi
  mid=(lo+hi)//2
  IF A[mid]==key RETURN mid
  IF A[mid] < key lo = mid+1 ELSE hi = mid-1
RETURN -1

8. BFS (graph)
QUEUE q; mark start visited; enqueue(start)
WHILE q not empty
  u = dequeue()
  FOR v in neighbors(u)
    IF not visited(v) mark visited(v); enqueue(v)

9. DFS (recursive)
FUNCTION dfs(u)
  mark visited(u)
  FOR v in neighbors(u)
    IF not visited(v) dfs(v)

10. FIBONACCI (iterative)
a=0; b=1
FOR i = 0 TO n-1
  a, b = b, a+b
RETURN a
